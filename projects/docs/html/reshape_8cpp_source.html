256 void Reshape::findMinAndMaxPositionFromSensorMask(HDF5Helper::Dataset *sensorMaskIndexDataset, HDF5Helper::Vector3D &min, HDF5Helper::Vector3D &max)
257 {
258  // Find min and max position from linear saved values
259  hsize_t index = 0;
260  hsize_t *data;
261  HDF5Helper::Vector3D offset;
262  HDF5Helper::Vector3D count;
263  HDF5Helper::Vector3D dstPos;
264  HDF5Helper::Vector4D nDims = sensorMaskIndexDataset->getFile()->getNdims();
265 
266  min = nDims;
267  max = HDF5Helper::Vector3D(0, 0, 0);
268 
269  for (hsize_t i = 0; i < sensorMaskIndexDataset->getNumberOfBlocks(); i++) {
270  sensorMaskIndexDataset->readBlock(i, offset, count, data);
271  for (hsize_t z = 0; z < count.z(); z++) {
272  for (hsize_t y = 0; y < count.y(); y++) {
273  for (hsize_t x = 0; x < count.x(); x++) {
274  // [x + y * count.x() + z * count.x() * count.y()] is indexed from 0, but value of index is from 1
275  index = data[x + y * count.x() + z * count.x() * count.y()] - 1;
276  HDF5Helper::convertlinearToMultiDim(index, dstPos, HDF5Helper::Vector3D(nDims));
277 
278  if (dstPos.x() < min.x()) min.x(dstPos.x());
279  if (dstPos.y() < min.y()) min.y(dstPos.y());
280  if (dstPos.z() < min.z()) min.z(dstPos.z());
281  if (dstPos.x() > max.x()) max.x(dstPos.x());
282  if (dstPos.y() > max.y()) max.y(dstPos.y());
283  if (dstPos.z() > max.z()) max.z(dstPos.z());
284  }
285  }
286  }
287  delete[] data; // !!
288  }
289 
290  std::cout << " min sensor mask point:\t" << min << std::endl;
