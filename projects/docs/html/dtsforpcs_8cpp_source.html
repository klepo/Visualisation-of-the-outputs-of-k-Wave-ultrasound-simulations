206 void DtsForPcs::findDatasetsForProcessing(const H5Helper::Group *group, const Settings *settings)
207 {
208  for (hsize_t i = 0; i < group->getNumObjs(); i++) {
209  H5G_obj_t type = group->getObjTypeByIdx(i);
210  std::string name = group->getObjNameByIdx(i);
211 
212  // Datasets
213  if (type == H5G_DATASET) {
214  if (group->getName() != "/")
215  name = group->getName() + "/" + name;
216  else
217  name = "/" + name;
218  // Filter by selected names
219  if (isFiltered(name, settings))
220  continue;
221 
222  H5Helper::Dataset *dataset = group->openDataset(i, Helper::enableDebugMsgs);
223  H5Helper::DatasetType datasetType = dataset->getType(sensorMaskSize);
224 
225  if (datasetType != H5Helper::DatasetType::UNKNOWN) {
226  bool tmpFlag = Helper::enableDebugMsgs;
227  if (settings->getFlagInfo()) {
228  Helper::enableDebugMsgs = true;
229  }
230  datasets.insert(H5Helper::PairOfDatasets(dataset->getName(), dataset));
231  Helper::printDebugMsg("----> " + dataset->getTypeString(datasetType) + " dataset: " + dataset->getName());
232  if (settings->getFlagInfo()) {
233  Helper::printDebugTwoColumnsTab("size", dataset->getDims());
234  Helper::printDebugTwoColumnsTab("chunk size", dataset->getChunkDims());
235  }
236  // Find min/max values
237  if (settings->getFlagFindMinMax()) {
238  dataset->findAndSetGlobalMinAndMaxValue(false, settings->getFlagLog());
239  }
240  // Print attributes
241  if (settings->getFlagInfo()) {
242  if (dataset->getNumAttrs() > 0) {
243  Helper::printDebugMsg2S("Attributes");
244  }
245  for (hsize_t i = 0; i < dataset->getNumAttrs(); i++) {
246  H5Helper::Attribute *attribute = dataset->getAttribute(i);
247  Helper::printDebugTwoColumnsTab(attribute->getName(), attribute->getStringValue());
248  delete attribute;
249  attribute = nullptr;
250  }
251  }
252  if (settings->getFlagInfo()) {
253  Helper::enableDebugMsgs = tmpFlag;
254  }
255  }
256  // Unknown type
257  else {
258  group->closeDataset(dataset, false);
259  }
260  }
261  // Groups
262  if (type == H5G_GROUP) {
263  H5Helper::Group *nextGroup = group->openGroup(i, false);
264  findDatasetsForProcessing(nextGroup, settings);
265  group->closeGroup(nextGroup, false);
266  }
267  }

276 bool DtsForPcs::isFiltered(std::string name, const Settings *settings)
277 {
278  if (settings->getFlagNames()) {
279  std::list<std::string> names = settings->getNames();
280  // Iterate
281  for (std::list<std::string>::const_iterator it = names.begin(); it != names.end(); ++it) {
282  std::string str = *it;
283  if (str.find("/") != 0)
284  str.insert(0, "/");
285  if (str == name)
286  return false;
287  if (str.find("/") != str.length() - 1)
288  str.append("/");
289  if (name.find(str) != std::string::npos) {
290  return false;
291  }
292  }
293  return true;
294  } else {
295  return false;
296  }
