205 void DtsForPcs::findDatasetsForProcessing(HDF5Helper::Group *group, Settings *settings)
206 {
207  for (hsize_t i = 0; i < group->getNumObjs(); i++) {
208  H5G_obj_t type = group->getObjTypeByIdx(i);
209  std::string name = group->getObjNameByIdx(i);
210 
211  // Datasets
212  if (type == H5G_DATASET) {
213  if (group->getName() != "/")
214  name = group->getName() + "/" + name;
215  else
216  name = "/" + name;
217  // Filter by selected names
218  if (isFiltered(name, settings))
219  continue;
220 
221  HDF5Helper::Dataset *dataset = group->openDataset(i);
222  HDF5Helper::DatasetType datasetType = dataset->getType(sensorMaskSize);
223 
224  if (datasetType != HDF5Helper::DatasetType::UNKNOWN) {
225  datasets.insert(HDF5Helper::PairOfDatasets(dataset->getName(), dataset));
226  std::cout << "----> " << dataset->getTypeString(datasetType) << " dataset: " << dataset->getName() << ", size: " << dataset->getDims() << ", chunk size: " << dataset->getChunkDims() << std::endl;
227  // Find min/max values
228  if (settings->getFlagFindMinMax()) {

231  // Print attributtes

264 bool DtsForPcs::isFiltered(std::string name, Settings *settings)
265 {
266  if (settings->getFlagNames()) {
267  std::list<std::string> names = settings->getNames();
268  // Iterate
269  for (std::list<std::string>::const_iterator it = names.begin(); it != names.end(); ++it) {
270  std::string str = *it;
271  if (str.find("/") != 0)
272  str.insert(0, "/");
273  if (str == name)
274  return false;
275  if (str.find("/") != str.length() - 1)
276  str.append("/");
277  if (name.find(str) != std::string::npos) {
278  return false;
279  }
280  }
281  return true;
282  } else {
283  return false;
284  }
