356 void ParamsDefinition::commandLineParse(int argc, char **argv)
357 {
358  // Params parsing
359  for (int i = 1; i < argc; i++) {
360  std::string flagString = argv[i];
361  // Remove '-'
362  flagString.erase(0, 1);
363 
364  // Compare flag
365  ParamsDefinition::Flags::iterator search = flags.find(flagString);
366  if (search == flags.end()) {
367  // Not found
368  std::cout << help << std::endl;
369  throw std::invalid_argument(argv[i]);
370  // Throw an exception
371  } else {
372  // Found
373  //std::string name = search->first;
374  ParamsDefinition::Flag *flag = &search->second;
375  flag->setEnabled(true);
376  size_t c = flag->getParams().getCount();
377  for (unsigned int j = 0; j < c; j++) {
378  ParamsDefinition::Type type = flag->getParams().getParamType(j);
379  i++;
380  if (argc <= i) {
381  std::cout << help << std::endl;
382  // Throw an exception
383  throw std::invalid_argument(flag->getName());
384  }
385  size_t size;
386  switch (type) {
387  case ParamsDefinition::INT: {
388  int number = static_cast<int>(std::stoi(argv[i], &size));
389  // Throw an exception
390  if (strlen(argv[i]) != size)
391  throw std::invalid_argument(argv[i]);
392  flag->setParam(j, static_cast<void *>(&number));
393  break;
394  }
395  case ParamsDefinition::LONGLONG: {
396  long long number = static_cast<long long>(std::stoll(argv[i], &size));
397  // Throw an exception
398  if (strlen(argv[i]) != size)
399  throw std::invalid_argument(argv[i]);
400  flag->setParam(j, static_cast<void *>(&number));
401  break;
402  }
403  case ParamsDefinition::UINT: {
404  unsigned int number = static_cast<unsigned int>(std::stoul(argv[i], &size));
405  // Throw an exception
406  if (strlen(argv[i]) != size)
407  throw std::invalid_argument(argv[i]);
408  flag->setParam(j, static_cast<void *>(&number));
409  break;
410  }
411  case ParamsDefinition::ULONGLONG: {
412  unsigned long long number = static_cast<unsigned long long>(std::stoull(argv[i], &size));
413  // Throw an exception
414  if (strlen(argv[i]) != size)
415  throw std::invalid_argument(argv[i]);
416  flag->setParam(j, static_cast<void *>(&number));
417  break;
418  }
419  case ParamsDefinition::FLOAT: {
420  float number = static_cast<float>(std::stof(argv[i], &size));
421  // Throw an exception
422  if (strlen(argv[i]) != size)
423  throw std::invalid_argument(argv[i]);
424  flag->setParam(j, static_cast<void *>(&number));
425  break;
426  }
427  case ParamsDefinition::DOUBLE: {
428  double number = static_cast<double>(std::stod(argv[i], &size));
429  // Throw an exception

431  throw std::invalid_argument(argv[i]);
432  flag->setParam(j, static_cast<void *>(&number));

434  }

437  // Throw an exception

439  throw std::invalid_argument(argv[i]);
440  flag->setParam(j, static_cast<void *>(&number));
441  break;
442  }
443  case ParamsDefinition::STRING: {
444  std::string string = argv[i];
445  flag->setParam(j, static_cast<void *>(&string));
446  break;
447  }

449  std::string myText(argv[i]);

451  std::string token;

458  }
459  break;
460  }
461  }
462  }
