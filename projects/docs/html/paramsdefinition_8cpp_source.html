527 void ParamsDefinition::commandLineParse(int argc, const char **argv)
528 {
529  // Params parsing
530  for (int i = 1; i < argc; i++) {
531  std::string flagString = argv[i];
532  // Remove '-'
533  flagString.erase(0, 1);
534 
535  // Compare flag
536  ParamsDefinition::Flags::iterator search = flags.find(flagString);
537  if (search == flags.end()) {
538  // Not found
539  std::cout << help << std::endl;
540  throw std::invalid_argument(argv[i]);
541  // Throw an exception
542  } else {
543  // Found
544  //std::string name = search->first;
545  ParamsDefinition::Flag *flag = &search->second;
546  flag->setEnabled(true);
547  size_t c = flag->getParams().getCount();
548  for (unsigned int j = 0; j < c; j++) {
549  ParamsDefinition::Type type = flag->getParams().getParamType(j);
550  i++;
551  if (argc <= i) {
552  std::cout << help << std::endl;
553  // Throw an exception
554  throw std::invalid_argument(flag->getName());
555  }
556  switch (type) {
557  case ParamsDefinition::INT: {
558  int number = ParamsDefinition::toInt(argv[i]);
559  flag->setParam(j, static_cast<void *>(&number));
560  break;
561  }
562  case ParamsDefinition::LONGLONG: {
563  long long number = ParamsDefinition::toLongLong(argv[i]);
564  flag->setParam(j, static_cast<void *>(&number));
565  break;
566  }
567  case ParamsDefinition::UINT: {
568  unsigned int number = ParamsDefinition::toUnsignedInt(argv[i]);
569  flag->setParam(j, static_cast<void *>(&number));
570  break;
571  }
572  case ParamsDefinition::ULONGLONG: {
573  unsigned long long number = ParamsDefinition::toUnsignedLongLong(argv[i]);
574  flag->setParam(j, static_cast<void *>(&number));
575  break;
576  }
577  case ParamsDefinition::ULONGLONG_SEPARATED: {
578  std::string myText(argv[i]);
579  std::istringstream iss(myText);
580  std::string token;
581  ParamsDefinition::VectorOfULongLongs vector;
582  while (getline(iss, token, ';')) {
583  unsigned long long number = ParamsDefinition::toUnsignedLongLong(token.c_str());
584  vector.push_back(number);
585  }
586  if (vector.size() > 0) {
587  flag->setParam(j, static_cast<void *>(&vector));
588  }
589  break;
590  }
591  case ParamsDefinition::FLOAT: {
592  float number = ParamsDefinition::toFloat(argv[i]);
593  flag->setParam(j, static_cast<void *>(&number));
594  break;
595  }
596  case ParamsDefinition::DOUBLE: {
597  double number = ParamsDefinition::toDouble(argv[i]);
598  flag->setParam(j, static_cast<void *>(&number));
599  break;
600  }
601  case ParamsDefinition::LONGDOUBLE: {
602  long double number = ParamsDefinition::toLongDouble(argv[i]);
603  flag->setParam(j, static_cast<void *>(&number));
604  break;
605  }
606  case ParamsDefinition::STRING: {
607  std::string string = argv[i];
608  flag->setParam(j, static_cast<void *>(&string));
609  break;
610  }
611  case ParamsDefinition::STRINGS_SEPARATED: {
612  std::string myText(argv[i]);
613  std::istringstream iss(myText);
614  std::string token;
615  ParamsDefinition::ListOfStrings list;
616  while (getline(iss, token, ';')) {
617  list.insert(list.begin(), token);
618  }
619  if (list.size() > 0) {
620  flag->setParam(j, static_cast<void *>(&list));
621  }
622  break;
623  }
624  }
625  }
626  }
627  }
628 }
