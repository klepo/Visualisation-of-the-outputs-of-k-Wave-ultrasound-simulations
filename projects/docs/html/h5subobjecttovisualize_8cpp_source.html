157 
158  // Set 3D frame size

169  originalSize.y(dataset->readAttributeI(HDF5Helper::SRC_SIZE_Y_ATTR, false));
170  originalSize.z(dataset->readAttributeI(HDF5Helper::SRC_SIZE_Z_ATTR, false));
171  originalFrameSize = originalSize;
172  }
173 
174  // Get chunk dimensions
175  chunkSize = dataset->getChunkDims();
176 
177  // Get global min/max values
178  dataset->findAndSetGlobalMinAndMaxValue();
179  hsize_t minValueIndex;
180  hsize_t maxValueIndex;
181  dataset->getGlobalMinValue(minValue, minValueIndex);
182  dataset->getGlobalMaxValue(maxValue, maxValueIndex);
183  dataset->getGlobalMinValue(originalMinValue, minValueIndex);
184  dataset->getGlobalMaxValue(originalMaxValue, maxValueIndex);
185 
186  HDF5Helper::convertlinearToMultiDim(minValueIndex, minValuePosition, dataset->getDims());
187  HDF5Helper::convertlinearToMultiDim(maxValueIndex, maxValuePosition, dataset->getDims());
188 
189 
190  if (type == dataset3D_t) {
191  // Default step
192  steps = 1;
193  if (dataset->hasAttribute(HDF5Helper::POSITION_X_ATTR) && dataset->hasAttribute(HDF5Helper::POSITION_Y_ATTR) && dataset->hasAttribute(HDF5Helper::POSITION_Z_ATTR)) {
194  originalPosition.z(dataset->readAttributeI(HDF5Helper::POSITION_Z_ATTR, false));
195  originalPosition.y(dataset->readAttributeI(HDF5Helper::POSITION_Y_ATTR, false));
196  originalPosition.x(dataset->readAttributeI(HDF5Helper::POSITION_X_ATTR, false));
197  position = originalPosition;
198 
199  frameSize = HDF5Helper::Vector3D(openedH5File->getNDims());;
200  originalFrameSize = frameSize;
201  }
202 
203  } else if (type == dataset4D_t) {
204 
205  frameSize = HDF5Helper::Vector3D(openedH5File->getNDims());;
206  originalFrameSize = frameSize;
207 
208  // Get position (was defined by sensor mask)
209  if (dataset->hasAttribute(HDF5Helper::POSITION_X_ATTR) && dataset->hasAttribute(HDF5Helper::POSITION_Y_ATTR) && dataset->hasAttribute(HDF5Helper::POSITION_Z_ATTR)) {
210  originalPosition.z(dataset->readAttributeI(HDF5Helper::POSITION_Z_ATTR, false));
211  originalPosition.y(dataset->readAttributeI(HDF5Helper::POSITION_Y_ATTR, false));

213  position = originalPosition;
214  }
215 
216  // Downsampled position
217  if (size != originalSize) {
218  float ratio = float(qMax(size.x(), qMax(size.y(), size.z()))) / qMax(originalSize.x(), qMax(originalSize.y(), originalSize.z()));
219  position.x(hsize_t(originalPosition.x() * ratio));
220  position.y(hsize_t(originalPosition.y() * ratio));
221  position.z(hsize_t(originalPosition.z() * ratio));
222

224  originalFrameSize.y(hsize_t(originalFrameSize.y() * ratio));
225  originalFrameSize.z(hsize_t(originalFrameSize.z() * ratio));
226  }
