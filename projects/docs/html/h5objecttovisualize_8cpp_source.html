161 void H5ObjectToVisualize::loadObjectData()
162 {
163  // Get chunk dimensions
164  chunkSize = dataset->getChunkDims();
165 
166  // Get global min/max values
167  dataset->findAndSetGlobalMinAndMaxValue();
168  hsize_t minValueIndex;
169  hsize_t maxValueIndex;
170  dataset->getGlobalMinValue(minValue, minValueIndex);
171  dataset->getGlobalMaxValue(maxValue, maxValueIndex);
172  originalMinValue = minValue;
173  originalMaxValue = maxValue;
174 
175  H5Helper::convertlinearToMultiDim(minValueIndex, minValuePosition, dataset->getDims());
176  H5Helper::convertlinearToMultiDim(maxValueIndex, maxValuePosition, dataset->getDims());
177 
178  // Size of dataset (can be downsampled)
179  size = H5Helper::Vector3D(dataset->getDims());
180  originalSize = size;
181 
182  // Set 3D frame size
183  frameSize = H5Helper::Vector3D(dataset->getFile()->getNDims());;
184  originalFrameSize = frameSize;
185 
186  // If masked
187  if (dataset->hasAttribute(H5Helper::POSITION_X_ATTR) && dataset->hasAttribute(H5Helper::POSITION_Y_ATTR) && dataset->hasAttribute(H5Helper::POSITION_Z_ATTR)) {
188  position.z(dataset->readAttributeI(H5Helper::POSITION_Z_ATTR, false));
189  position.y(dataset->readAttributeI(H5Helper::POSITION_Y_ATTR, false));
190  position.x(dataset->readAttributeI(H5Helper::POSITION_X_ATTR, false));
191  originalPosition = position;
192  } else if (type == H5OpenedFile::DATASET_4D) {
193  // Try to get position from sensor_mask_corners dataset
194  if (dataset->getFile()->objExistsByName(H5Helper::SENSOR_MASK_CORNERS_DATASET)) {
195  H5Helper::Dataset *sensorMaskCornersDataset = dataset->getFile()->openDataset(H5Helper::SENSOR_MASK_CORNERS_DATASET);
196  hsize_t *sensorMaskCornersData = nullptr;
197  sensorMaskCornersDataset->readDataset(sensorMaskCornersData);
198  // Name of the dataset to index
199  hsize_t i = hsize_t(std::stoi(dataset->getOnlyName()));
200  dataset->findAndSetGlobalMinAndMaxValue();
201  dataset->setAttribute(H5Helper::POSITION_Z_ATTR, sensorMaskCornersData[(i - 1) * 6 + 2] - 1);
202  dataset->setAttribute(H5Helper::POSITION_Y_ATTR, sensorMaskCornersData[(i - 1) * 6 + 1] - 1);
203  dataset->setAttribute(H5Helper::POSITION_X_ATTR, sensorMaskCornersData[(i - 1) * 6 + 0] - 1);
204 
205  position.z(sensorMaskCornersData[(i - 1) * 6 + 2] - 1);
206  position.y(sensorMaskCornersData[(i - 1) * 6 + 1] - 1);
207  position.x(sensorMaskCornersData[(i - 1) * 6 + 0] - 1);
208  originalPosition = position;
209  delete[] sensorMaskCornersData;
210  sensorMaskCornersData = nullptr;
211  dataset->getFile()->closeDataset(sensorMaskCornersDataset);
212  }
213  }
214 
215  // If downsampled
216  if (dataset->hasAttribute(H5Helper::SRC_SIZE_X_ATTR)
217  && dataset->hasAttribute(H5Helper::SRC_SIZE_Y_ATTR)
218  && dataset->hasAttribute(H5Helper::SRC_SIZE_Z_ATTR)
219  && dataset->hasAttribute(H5Helper::SRC_DATASET_NAME_ATTR)
220  ) {
221  originalSize.x(dataset->readAttributeI(H5Helper::SRC_SIZE_X_ATTR, false));
222  originalSize.y(dataset->readAttributeI(H5Helper::SRC_SIZE_Y_ATTR, false));
223  originalSize.z(dataset->readAttributeI(H5Helper::SRC_SIZE_Z_ATTR, false));
224 
225  if (dataset->hasAttribute(H5Helper::SRC_POSITION_Z_ATTR)
226  && dataset->hasAttribute(H5Helper::SRC_POSITION_Y_ATTR)
227  && dataset->hasAttribute(H5Helper::SRC_POSITION_X_ATTR)
228  ) {
229  originalPosition.x(dataset->readAttributeI(H5Helper::SRC_POSITION_X_ATTR, false));
230  originalPosition.y(dataset->readAttributeI(H5Helper::SRC_POSITION_Y_ATTR, false));
231  originalPosition.z(dataset->readAttributeI(H5Helper::SRC_POSITION_Z_ATTR, false));
232  }
233 
234  float ratio = float(qMax(size.x(), qMax(size.y(), size.z()))) / qMax(originalSize.x(), qMax(originalSize.y(), originalSize.z()));
235  frameSize.x(qRound(frameSize.x() * ratio));
236  frameSize.y(qRound(frameSize.y() * ratio));
237  frameSize.z(qRound(frameSize.z() * ratio));
238  }
239 
240  if (type == H5OpenedFile::DATASET_4D) {
241  // Get number of steps
242  steps = H5Helper::Vector4D(dataset->getDims()).w();
243  }
244 
245  if (dataset->getType() == H5Helper::DatasetType::CUBOID_ATTR_C || dataset->getType() == H5Helper::DatasetType::CUBOID_C) {
246  hsize_t mos = dataset->hasAttribute(H5Helper::C_MOS_ATTR) ? dataset->readAttributeI(H5Helper::C_MOS_ATTR) : 1;
247  hsize_t harmonics = dataset->hasAttribute(H5Helper::C_HARMONICS_ATTR) ? dataset->readAttributeI(H5Helper::C_HARMONICS_ATTR) : 1;
248  compressHelper = new H5Helper::CompressHelper(dataset->readAttributeI(H5Helper::C_PERIOD_ATTR), mos, harmonics);
249 
250  size.x(size.x() / compressHelper->getStride());
251  originalSize.x(originalSize.x() / compressHelper->getStride());
252 
253  steps = (steps + 2) * compressHelper->getOSize();
254  H5Helper::convertlinearToMultiDim(minValueIndex, minValuePosition, H5Helper::Vector4D(steps, size));
255  H5Helper::convertlinearToMultiDim(maxValueIndex, maxValuePosition, H5Helper::Vector4D(steps, size));
256 
257  threadXY->setCompressHelper(compressHelper);
258  threadXZ->setCompressHelper(compressHelper);
259  threadYZ->setCompressHelper(compressHelper);
260  thread3D->setCompressHelper(compressHelper);
261  }
