1000 void H5ObjectToVisualize::loadObjectData()
1001 {
1002  // Get chunk dimensions
1003  chunkSize = dataset->getChunkDims();
1004 
1005  // Get global min/max values
1006  dataset->findAndSetGlobalMinAndMaxValue();
1007  hsize_t minValueIndex;
1008  hsize_t maxValueIndex;
1009  dataset->getGlobalMinValue(minValue, minValueIndex);
1010  dataset->getGlobalMaxValue(maxValue, maxValueIndex);
1011  originalMinValue = minValue;
1012  originalMaxValue = maxValue;
1013 
1014  H5Helper::convertlinearToMultiDim(minValueIndex, minValuePosition, dataset->getDims());
1015  H5Helper::convertlinearToMultiDim(maxValueIndex, maxValuePosition, dataset->getDims());
1016 
1017  // Size of dataset (can be downsampled)
1018  size = H5Helper::Vector3D(dataset->getDims());
1019  originalSize = size;
1020 
1021  // Set 3D frame size
1022  frameSize = H5Helper::Vector3D(dataset->getFile()->getNDims());;
1023  originalFrameSize = frameSize;
1024 
1025  // If masked
1026  if (dataset->hasAttribute(H5Helper::POSITION_X_ATTR) && dataset->hasAttribute(H5Helper::POSITION_Y_ATTR) && dataset->hasAttribute(H5Helper::POSITION_Z_ATTR)) {
1027  position.z(dataset->readAttributeI(H5Helper::POSITION_Z_ATTR, false));
1028  position.y(dataset->readAttributeI(H5Helper::POSITION_Y_ATTR, false));
1029  position.x(dataset->readAttributeI(H5Helper::POSITION_X_ATTR, false));
1030  originalPosition = position;
1031  } else if (type == H5OpenedFile::DATASET_4D) {
1032  // Try to get position from sensor_mask_corners dataset
1033  if (dataset->getFile()->objExistsByName(H5Helper::SENSOR_MASK_CORNERS_DATASET)) {
1034  H5Helper::Dataset *sensorMaskCornersDataset = dataset->getFile()->openDataset(H5Helper::SENSOR_MASK_CORNERS_DATASET);
1035  hsize_t *sensorMaskCornersData = nullptr;
1036  sensorMaskCornersDataset->readDataset(sensorMaskCornersData);
1037  // Name of the dataset to index
1038  hsize_t i = hsize_t(std::stoi(dataset->getOnlyName()));
1039  dataset->findAndSetGlobalMinAndMaxValue();
1040  dataset->setAttribute(H5Helper::POSITION_Z_ATTR, sensorMaskCornersData[(i - 1) * 6 + 2] - 1);
1041  dataset->setAttribute(H5Helper::POSITION_Y_ATTR, sensorMaskCornersData[(i - 1) * 6 + 1] - 1);
1042  dataset->setAttribute(H5Helper::POSITION_X_ATTR, sensorMaskCornersData[(i - 1) * 6 + 0] - 1);
1043 
1044  position.z(sensorMaskCornersData[(i - 1) * 6 + 2] - 1);
1045  position.y(sensorMaskCornersData[(i - 1) * 6 + 1] - 1);
1046  position.x(sensorMaskCornersData[(i - 1) * 6 + 0] - 1);
1047  originalPosition = position;
1048  delete[] sensorMaskCornersData;
1049  sensorMaskCornersData = nullptr;
1050  dataset->getFile()->closeDataset(sensorMaskCornersDataset);
1051  }
1052  }
1053 
1054  // If downsampled
1055  if (dataset->hasAttribute(H5Helper::SRC_SIZE_X_ATTR)
1056  && dataset->hasAttribute(H5Helper::SRC_SIZE_Y_ATTR)
1057  && dataset->hasAttribute(H5Helper::SRC_SIZE_Z_ATTR)
1058  && dataset->hasAttribute(H5Helper::SRC_DATASET_NAME_ATTR)
1059  ) {
1060  originalSize.x(dataset->readAttributeI(H5Helper::SRC_SIZE_X_ATTR, false));
1061  originalSize.y(dataset->readAttributeI(H5Helper::SRC_SIZE_Y_ATTR, false));
1062  originalSize.z(dataset->readAttributeI(H5Helper::SRC_SIZE_Z_ATTR, false));
1063 
1064  if (dataset->hasAttribute(H5Helper::SRC_POSITION_Z_ATTR)
1065  && dataset->hasAttribute(H5Helper::SRC_POSITION_Y_ATTR)
1066  && dataset->hasAttribute(H5Helper::SRC_POSITION_X_ATTR)
1067  ) {
1068  originalPosition.x(dataset->readAttributeI(H5Helper::SRC_POSITION_X_ATTR, false));
1069  originalPosition.y(dataset->readAttributeI(H5Helper::SRC_POSITION_Y_ATTR, false));
1070  originalPosition.z(dataset->readAttributeI(H5Helper::SRC_POSITION_Z_ATTR, false));
1071  }
1072 
1073  float ratio = float(qMax(size.x(), qMax(size.y(), size.z()))) / qMax(originalSize.x(), qMax(originalSize.y(), originalSize.z()));
1074  frameSize.x(qRound(frameSize.x() * ratio));
1075  frameSize.y(qRound(frameSize.y() * ratio));
1076  frameSize.z(qRound(frameSize.z() * ratio));
1077  }
1078 
1079  if (type == H5OpenedFile::DATASET_4D) {
1080  // Get number of steps
1081  steps = H5Helper::Vector4D(dataset->getDims()).w();
1082  }
1083 
1084  if (dataset->getType() == H5Helper::DatasetType::CUBOID_ATTR_C || dataset->getType() == H5Helper::DatasetType::CUBOID_C) {
1085  hsize_t mos = dataset->hasAttribute(H5Helper::C_MOS_ATTR) ? dataset->readAttributeI(H5Helper::C_MOS_ATTR) : 1;
1086  hsize_t harmonics = dataset->hasAttribute(H5Helper::C_HARMONICS_ATTR) ? dataset->readAttributeI(H5Helper::C_HARMONICS_ATTR) : 1;
1087  compressHelper = new H5Helper::CompressHelper(dataset->readAttributeI(H5Helper::C_PERIOD_ATTR), mos, harmonics);
1088 
1089  size.x(size.x() / compressHelper->getStride());
1090  originalSize.x(originalSize.x() / compressHelper->getStride());
1091 
1092  steps = (steps + 2) * compressHelper->getOSize();
1093  H5Helper::convertlinearToMultiDim(minValueIndex, minValuePosition, H5Helper::Vector4D(steps, size));
1094  H5Helper::convertlinearToMultiDim(maxValueIndex, maxValuePosition, H5Helper::Vector4D(steps, size));
1095 
1096  threadXY->setCompressHelper(compressHelper);
1097  threadXZ->setCompressHelper(compressHelper);
1098  threadYZ->setCompressHelper(compressHelper);
1099  thread3D->setCompressHelper(compressHelper);
1100  }
