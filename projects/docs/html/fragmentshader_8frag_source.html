55 bool isTrimmed(float value) {
56  if (uTrim) {
57  if (uMin <= uMax) {
58  if (value > uMax) {
59  return true;
60  }
61  if (value < uMin) {
62  return true;
63  }
64  } else {
65  if (value < uMax) {
66  return true;
67  }
68  if (value > uMin) {
69  return true;
70  }
71  }
72  }
73  return false;
74 }

76 vec4 computeColor(float texel)
77 {
78  if (isTrimmed(texel)) {
79  return vec4(0, 0, 0, 0);
80  }
81  texel = ((texel - uMin) * 1.0f) / (uMax - uMin);
82  vec4 color = texture(uColormap, texel);
83  vec4 opacity = texture(uOpacity, texel);
84  return vec4(color.rgb, opacity);
85 }

87 void main() {
88  if (uFrame) {
89  outColor = uFrameColor;
90  } else if (uVolumeRenderingBox) {
91  outColor = vec4(vTextureCoordBox.stp, 1.0f);
92  } else if (uSlices) {
93  if (uXYBorder) {
94  outColor = vec4(0.0f, 0.0f, 0.8f, 1.0f);
95  } else if (uXZBorder) {
96  outColor = vec4(0.0f, 0.8f, 0.0f, 1.0f);
97  } else if (uYZBorder) {
98  outColor = vec4(0.8f, 0.0f, 0.0f, 1.0f);
99  } else {
100  float value = texture(uSlice, vTextureCoord).r;
101  if (isTrimmed(value)) {
102  discard;
103  return;
104  }
105  value = ((value - uMin) * 1.0f) / (uMax - uMin);
106  outColor = vec4(texture(uColormap, value).rgb, 1.0f);
107  //outColor = vec4(vTextureCoordBox.stp, 1.0f);
108  }
109  } else if (uVolumeRendering) {
110  vec2 coords = vec2((gl_FragCoord.x / uWidth), (gl_FragCoord.y / uHeight));
111  vec3 backPoint = vec3(texture(uBoxBackSampler, coords));
112  vec3 frontPoint = vec3(texture(uBoxFrontSampler, coords));
113  //vec3 frontPoint = vTextureCoordBox.stp;
114 
115  float depth = distance(frontPoint, backPoint);
116 
117  vec3 path = normalize(backPoint - frontPoint) / uSteps;
118  vec3 point = frontPoint;
119  float texel = 0.0f;
120 
121  int i = 0;
122  int it = int(depth / length(path));
123 
124  //outColor = vec4(vec3(length(path)), 1.0f);
125  //outColor = vec4(vec3(depth), 1.0f);
126  //outColor = vec4(backPoint, 1.0f);
127  //outColor = vec4(frontPoint, 1.0f);
128  //return;
129 
130  vec4 cOut = vec4(0, 0, 0, 0);
131 
132  outColor = cOut;
133 
134  for (i = 0; i < it; i++) {
135  if (uMode == 1) { // Maximum intensity projection
136  texel = max(texture(uVolume, point).r, texel);
137  } else if (uMode == 2) { // Minimum intensity projection
138  texel = min(texture(uVolume, point).r, texel);
139  } else if (uMode == 0) { // Accumulation
140  vec4 cIn = computeColor(texture(uVolume, point).r);
141  //if (cIn.a == 0.0f)
142  // continue;
143  cIn.a = 50 * cIn.a / uSteps;
144  cOut.rgb = cOut.rgb + (1 - cOut.a) * cIn.rgb * cIn.a;
145  cOut.a = cOut.a + (1 - cOut.a) * cIn.a;
146  if (cOut.a >= 1.0f)
147  break;
148  } else if (uMode == 3) { // Average intensity projection
149  texel += texture(uVolume, point).r;
150  } else { // Unknown mode
151  break;
152  }
153  point += path;
154  }
155 
156  if (uMode == 3) { // Average intensity projection
157  texel = texel / i;
158  }
159 
160  if (uMode == 0) { // Accumulation
161  outColor = cOut;
162  } else {
163  outColor = computeColor(texel);
164  }
165 
166  } else {
167  outColor = vec4(0.5f, 0.5f, 0.5f, 1.0f);
168  }
169 }
