103  outputDims[0] = outputSteps;
104  outputDims[3] /= compressHelper->getHarmonics() * 2;
105  stepSize = dims[1] * dims[2] * dims[3];
106  outputStepSize = outputDims[1] * outputDims[2] * outputDims[3];
107  } else if (dims.getLength() == 3) { // 3D dataset (defined by sensor mask)

109  outputSteps = (steps + 2) * oSize;
110  outputDims[1] = outputSteps;
111  outputDims[2] /= compressHelper->getHarmonics() * 2;
112  stepSize = dims[2];
113  outputStepSize = outputDims[2];
114  } else { // Something wrong.

122  chunkDims[3] /= compressHelper->getHarmonics() * 2;
123  } else if (dims.getLength() == 3) { // 3D dataset (defined by sensor mask)
124  chunkDims[2] /= compressHelper->getHarmonics() * 2;
125  } else { // Something wrong.

155  if (srcDataset->getFile()->getNumberOfElmsToLoad() >= stepSize * 2 + outputStepSize) {
156  // Complex buffers for last coefficients
157  H5Helper::floatC *cC = new H5Helper::floatC[stepSize / 2]();
158  H5Helper::floatC *lC = new H5Helper::floatC[stepSize / 2]();
159

162

165 
166  hsize_t step = 0;
167  hsize_t stepToWrite = 0;

170  for (hsize_t i = 0; i < srcDataset->getNumberOfBlocks(); i++) {
171  srcDataset->readBlock(i, offset, count, dataC, log);
172 
173  hsize_t framesCount;
174  hsize_t framesOffset;
175 
176  if (dims.getLength() == 4) { // 4D dataset
177  framesCount = count[0];
178  framesOffset = offset[0];
179  } else { // 3D dataset
180  framesCount = count[1];
181  framesOffset = offset[1];
182  }
183 
184  // Because of last coefficients duplication
185  if (framesOffset + framesCount == steps)
186  framesCount += 2;
187 
188  // For every frame
189  for (hsize_t frame = 0; frame < framesCount; frame++) {
190  hsize_t framesOffsetGlobal = framesOffset + frame;
191  hsize_t frameOffset = frame * stepSize;
192  // Decode steps
193  for (hsize_t stepLocal = 0; stepLocal < oSize; stepLocal++) {
194  hsize_t stepOffset = step * outputStepSize;
195  hsize_t stepsToWriteOffset = stepToWrite * outputStepSize;
196 
197  // For every coefficient (space point) in frame
198  //#pragma omp parallel for
199  for (hssize_t p = 0; p < hssize_t(outputStepSize); p++) {
200  hsize_t pOffset = compressHelper->getHarmonics() * hsize_t(p);
201  hsize_t sP = stepsToWriteOffset + p;
202  data[sP] = 0;
203 
204  // For every hamornics
205  for (hsize_t ih = 0; ih < compressHelper->getHarmonics(); ih++) {
206  hsize_t pH = pOffset + ih;
207 
208  if (stepLocal == 0) {
209  hsize_t pHC = 2 * pOffset + 2 * ih;
210 
211  // Save last coefficients

213 
214  // Copy first coefficients

217 
218  // Don't load last two coefficients (duplicate)
219  if (framesOffsetGlobal < steps - 1) {
220  hsize_t fPHC = frameOffset + pHC;
221  // Read coefficient

223  }
224  }
225

228  data[sP] += real(cC[pH] * compressHelper->getBE()[sH]) + real(lC[pH] * compressHelper->getBE_1()[sH]);
229  }
230 
231  // Min/max values
232  H5Helper::checkOrSetMinMaxValue(minV, maxV, data[sP], minVIndex, maxVIndex, stepOffset + hsize_t(p));
233  }
234

236  stepToWrite++;
237 
238  if (step % stepsToWrite == 0 || step == outputSteps) {

241 
242  if (dims.getLength() == 4) // 4D dataset
243  dstDataset->writeDataset(H5Helper::Vector4D(step - stepToWrite, 0, 0, 0), H5Helper::Vector4D(stepToWrite, outputDims[1], outputDims[2], outputDims[3]), data);
244  else if (dims.getLength() == 3) // 3D dataset
245  dstDataset->writeDataset(H5Helper::Vector3D(0, step - stepToWrite, 0), H5Helper::Vector3D(1, stepToWrite, outputDims[2]), data);
246 
247  if (log)
248  Helper::printDebugMsg("saved");

251  }
252  }
253  }
254  }
255  // Delete buffers
