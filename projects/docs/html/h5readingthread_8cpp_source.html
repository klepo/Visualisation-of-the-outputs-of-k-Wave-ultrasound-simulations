231  Request *r = nullptr;
232  // Queue mutex
233  queueMutex.lock();
234  if (!queue.isEmpty())
235  // Get waiting request
236  r = queue.dequeue();

238  queueMutex.unlock();

240  }

243  bool log = false;
244

252  if (compressHelper) {
253  hsize_t xStride = compressHelper->getStride();
254  hsize_t oSize = compressHelper->getOSize();
255  hsize_t step = H5Helper::Vector4D(r->offset).t();
256  hsize_t stepC = step / oSize;
257  localStep = step - stepC * oSize;
258  hsize_t steps = H5Helper::Vector4D(r->dataset->getDims()).t();
259  H5Helper::Vector4D offsetLC = r->offset;

261  H5Helper::Vector4D count = r->count;
262  count.x(count.x() * xStride);
263 
264  offsetLC.t((stepC) - 1);
265  if (hssize_t(stepC) - 1 < 0)
266  offsetLC.t(0);
267  if (offsetLC.t() >= steps)
268  offsetLC.t(steps - 1);
269  offsetLC.x(offsetLC.x() * xStride);
270 
271  offsetCC.t(stepC);
272  if (offsetCC.t() >= steps)
273  offsetCC.t(steps - 1);
274  offsetCC.x(offsetCC.x() * xStride);
275 
276  if (!this->dataLC || this->offsetLC != offsetLC) {
277  r->dataset->readDataset(offsetLC, count, this->dataLC, log);
278  this->offsetLC = offsetLC;
279  }
280 
281  if (!this->dataCC || this->offsetCC != offsetCC) {
282  r->dataset->readDataset(offsetCC, count, this->dataCC, log);
283  this->offsetCC = offsetCC;
284  }
285 
286  if (!r->full) {
287  if (!r->extData) {
288  r->data = new float[r->count.getSize()]();
289  }
290 
291 #pragma omp parallel for

293  r->data[p] = compressHelper->computeTimeStep(&this->dataCC[p * xStride], &this->dataLC[p * xStride], localStep);
294  }
295  }
296  } else {

298  r->data = new float[r->count.getSize()]();
299  }

301  }
