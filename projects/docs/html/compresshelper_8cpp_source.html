450 void CompressHelper::generateBE(hsize_t ih, hsize_t bSize, hsize_t oSize, const float *b, const floatC *e, floatC *bE, floatC *bE_1, bool normalize) const
451 {
452  for (hsize_t x = 0; x < hsize_t(bSize); x++) {

454  bE[hx] = b[x] * e[hx];
