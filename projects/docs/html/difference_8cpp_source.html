51  H5Helper::Dataset *datasetOriginal = it->second;
52  H5Helper::DatasetType datasetType = datasetOriginal->getType(sensorMaskSize);
53  if (checkDatasetType(datasetType, types)) {
54  H5Helper::Dataset *datasetDecoded = nullptr;
55 
56  for (H5Helper::MapOfDatasetsIt it2 = map.begin(); it2 != map.end(); ++it2) {
57  H5Helper::Dataset *dataset = it2->second;
58  H5Helper::DatasetType datasetDecodedType = dataset->getType(sensorMaskSize);
59 
60  if ((datasetDecodedType == H5Helper::DatasetType::TIME_STEPS_D_INDEX && datasetType == H5Helper::DatasetType::TIME_STEPS_INDEX)
61  || (datasetDecodedType == H5Helper::DatasetType::CUBOID_D && datasetType == H5Helper::DatasetType::CUBOID)
62  || (datasetDecodedType == H5Helper::DatasetType::CUBOID_ATTR_D && datasetType == H5Helper::DatasetType::CUBOID_ATTR)
63  ) {
64  if (datasetOriginal->getName() == dataset->readAttributeS(H5Helper::SRC_DATASET_NAME_ATTR, false)) {
65  datasetDecoded = dataset;
66  break;
67  }
68  }
69  }
