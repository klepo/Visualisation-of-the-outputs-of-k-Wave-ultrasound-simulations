49  int count = 0;
50  for (HDF5Helper::MapOfDatasetsIt it = map.begin(); it != map.end(); ++it) {
51  HDF5Helper::Dataset *datasetOriginal = it->second;
52  HDF5Helper::DatasetType datasetType = datasetOriginal->getType(sensorMaskSize);
53 
54  if (checkDatasetType(datasetType, types)) {
55  HDF5Helper::Dataset *datasetDecoded = 0;
56 
57  for (HDF5Helper::MapOfDatasetsIt it2 = map.begin(); it2 != map.end(); ++it2) {
58  HDF5Helper::Dataset *dataset = it2->second;
59  HDF5Helper::DatasetType datasetDecodedType = dataset->getType(sensorMaskSize);
60  if ((datasetDecodedType == HDF5Helper::DatasetType::TIME_STEPS_D_MASK && datasetType == HDF5Helper::DatasetType::TIME_STEPS_MASK)
61  || (datasetDecodedType == HDF5Helper::DatasetType::CUBOID_D && datasetType == HDF5Helper::DatasetType::CUBOID)
62  || (datasetDecodedType == HDF5Helper::DatasetType::CUBOID_ATTR_D && datasetType == HDF5Helper::DatasetType::CUBOID_ATTR)
63  ) {
64  if (datasetOriginal->getName() == dataset->readAttributeS(HDF5Helper::SRC_DATASET_NAME_ATTR, false)) {
65  datasetDecoded = dataset;
66  break;
67  }
68  }
69  }
70 
71  if (datasetDecoded) {
72  std::cout << "Subtraction of datasets " << datasetOriginal->getName() << " and " << datasetDecoded->getName() << std::endl;
73  subtractDatasets(datasetOriginal, datasetDecoded);
74  std::cout << "Subtraction of datasets done" << std::endl << std::endl;
75  count++;
76  }
77  }
78  }
79

82  }

102 
103  // Variables for block reading
104  float *dataO = 0;
105  float *dataD = 0;
106  HDF5Helper::Vector offset;
107  HDF5Helper::Vector count;

110  bool first = true;
111  double sum = 0.0;
112  double sumO = 0.0;
113  double sum2 = 0.0;
114 
115  datasetDecoded->setNumberOfElmsToLoad(datasetOriginal->getNumberOfElmsToLoad());
116  // Reading and substraction
117  for (hsize_t i = 0; i < datasetOriginal->getNumberOfBlocks(); i++) {
118  // First read decoded -> it is larger
119  datasetDecoded->readBlock(i, offset, count, dataD);
120  datasetOriginal->readBlock(i, offset, count, dataO);
121  // The count is from original dataset and is smaller for last step than is in the decoded dataset
122 
123  #pragma omp parallel for reduction(+ : sum, sumO, sum2)
124  for (hssize_t i = 0; i < hssize_t(count.getSize()); i++) {
125  dataD[i] = dataO[i] - dataD[i];
126  sum += double(abs(dataD[i]));
127  sumO += double(abs(dataO[i]));
128  sum2 += double((dataD[i] * dataD[i]));
129  // Min/max values
130  HDF5Helper::checkOrSetMinMaxValue(first, minV, maxV, dataD[i]);
131 
132  hsize_t linearOffset;
133  convertMultiDimToLinear(offset, linearOffset, datasetDecoded->getDims());
134 
135  #pragma omp critical
136  {
137  if (dataD[i] == minV)
138  minVIndex = linearOffset + i;
139  if (dataD[i] == maxV)
140  maxVIndex = linearOffset + i;
141  }
142  }
143 
144  dstDataset->writeDataset(offset, count, dataD, true);
145  delete[] dataD;

147  }
148 
149  // Copy attributes

151 
152  float minVO;
153  float maxVO;
154  hsize_t minVOIndex;
155  hsize_t maxVOIndex;
156  datasetOriginal->getGlobalMinValue(minVO, minVOIndex);
157  datasetOriginal->getGlobalMaxValue(maxVO, maxVOIndex);
158  float maxError = std::max(abs(minV), abs(maxV));
159  double meanError = sum / double(outputDims.getSize());
160  float maxValue = std::max(abs(minVO), abs(maxVO));

176  dstDataset->setAttribute("mean_error_value", meanError);
177  dstDataset->setAttribute("max_error", (maxError / maxValue) * 100);
178  dstDataset->setAttribute("max_error_value", maxError);
179  dstDataset->setAttribute("mean_squared_error", float(mse));
180  dstDataset->setAttribute("snr", float(sumO / mse));
