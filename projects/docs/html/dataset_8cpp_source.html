1322 void Dataset::initBlockReading()
1323 {
1324  hsize_t prod = 1;
1325  blockDims = Vector(dims.getLength(), 1);
1326  lastBlockDims = Vector(dims.getLength(), 1);
1327  numberOfBlocksInDims = dims;
1328 
1329  lastBlockCount = 1;
1330  bool diffSizeFlag = false;
1331  realNumberOfElementsToLoad = getNumberOfElmsToLoad();
1332 
1333  for (hsize_t i = dims.getLength(); i > 0; i--) {
1334  hsize_t j = i - 1;
1335  hsize_t newProd = prod * dims[j];
1336  if (newProd > getNumberOfElmsToLoad()) {
1337  blockDims[j] = getNumberOfElmsToLoad() / prod;
1338  lastBlockDims[j] = blockDims[j];
1339  numberOfBlocksInDims[j] = dims[j] / blockDims[j];
1340  lastBlockCount = numberOfBlocksInDims[j];
1341 
1342  if (dims[j] % blockDims[j]) {
1343  lastBlockDims[j] = dims[j] % blockDims[j];
1344  diffSizeFlag = true;
1345  lastBlockCount++;
1346  }
1347  realNumberOfElementsToLoad = prod * blockDims[j];
1348  numberOfBlocksInDims[j] = lastBlockCount;
1349 
1350  break;
1351  }
1352  prod = newProd;
1353  blockDims[j] = dims[j];
1354  lastBlockDims[j] = dims[j];
1355  numberOfBlocksInDims[j] = 1;
1356  }
1357 
1358  numberOfBlocks = 1;
1359  for (hsize_t i = 0; i < dims.getLength(); i++) {
1360  hsize_t p = dims[i] / blockDims[i];
1361  if (dims[i] % blockDims[i])
1362  p++;
1363  numberOfBlocks *= p;
1364  }
1365 }
