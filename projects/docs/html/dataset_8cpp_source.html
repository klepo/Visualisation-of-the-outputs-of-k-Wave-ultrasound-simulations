1293 void Dataset::initBlockReading()
1294 {
1295  hsize_t prod = 1;
1296  blockDims = Vector(dims.getLength(), 1);
1297  lastBlockDims = Vector(dims.getLength(), 1);
1298  numberOfBlocksInDims = dims;
1299 
1300  lastBlockCount = 1;
1301  bool diffSizeFlag = false;
1302  realNumberOfElementsToLoad = getNumberOfElmsToLoad();
1303 
1304  for (hsize_t i = dims.getLength(); i > 0; i--) {
1305  hsize_t j = i - 1;
1306  hsize_t newProd = prod * dims[j];
1307  if (newProd > getNumberOfElmsToLoad()) {
1308  blockDims[j] = getNumberOfElmsToLoad() / prod;
1309  lastBlockDims[j] = blockDims[j];
1310  numberOfBlocksInDims[j] = dims[j] / blockDims[j];
1311  lastBlockCount = numberOfBlocksInDims[j];
1312 
1313  if (dims[j] % blockDims[j]) {
1314  lastBlockDims[j] = dims[j] % blockDims[j];
1315  diffSizeFlag = true;
1316  lastBlockCount++;
1317  }
1318  realNumberOfElementsToLoad = prod * blockDims[j];
1319  numberOfBlocksInDims[j] = lastBlockCount;
1320 
1321  break;
1322  }
1323  prod = newProd;
1324  blockDims[j] = dims[j];
1325  lastBlockDims[j] = dims[j];
1326  numberOfBlocksInDims[j] = 1;
1327  }
1328 
1329  numberOfBlocks = 1;
1330  for (hsize_t i = 0; i < dims.getLength(); i++) {
1331  hsize_t p = dims[i] / blockDims[i];
