425 void GWindow::setLoaded(Request *request)
426 {
427  if (selectedDataset != request->dataset)
428  return;
429 
430  texture3DInitialized = false;
431  HDF5Helper::Vector3D offset = request->offset;
432  HDF5Helper::Vector3D count = request->count;
433 
434  // Set 3D data to 3D texture
435  glBindTexture(GL_TEXTURE_3D, texture);
436  glPixelStorei(GL_UNPACK_ALIGNMENT, 1);
437  glTexSubImage3D(GL_TEXTURE_3D, 0, int(offset.x()), int(offset.y()), int(offset.z()), int(count.x()), int(count.y()), int(count.z()), GL_RED, GL_FLOAT, request->data);
438  glBindTexture(GL_TEXTURE_3D, 0);
439 
440  // Last block of 3D data
441  if (offset.z() + count.z() == hsize_t(imageSize.z())) {
442  texture3DInitialized = true;
443  //changeColormap(colormap);
444  renderLater();
445  emit loaded(selectedDataset->getName());
446  }
447 
448  thread->deleteDoneRequest(request);
449 }

690  position.setX(position.x() - float(lastPosRl.x() - currentPosRl.x()));

701 
702  // Zoom
703  if (this->getWheelDelta() > 0)
704  zoom *= 1.0f / 1.1f;
705  else if (this->getWheelDelta() < 0)
706  zoom *= 1.1f;
707 
708  // Projection
709  QMatrix4x4 projectionMatrix;
710  float ratio = float(width()) / float(height());
711 
712  if (orthogonal) {
713  if (zoom > 200.0f) {
714  zoom = 200.0f;
715  }
716  if (zoom < 0.00001f) {
717  zoom = 0.00001f;
718  }
719  projectionMatrix.translate(position.x(), position.y(), 0);
720  projectionMatrix.ortho(-ratio * zoom, ratio * zoom, -zoom, zoom, -1.0f, 1000.0f);
721 
722  } else {
723  float angle = zoom * 45.0f;
724  if (angle > 179.9999f) {
725  angle = 179.9999f;
726  zoom = 179.9999f / 45.0f;
727  }
728  if (angle < 0.00001f) {

733  projectionMatrix.perspective(angle, ratio, 0.1f, 100.0f);
734  projectionMatrix.translate(0, 0, -2.0f);
735  }

739  matrix = projectionMatrix * rotateXMatrix * rotateYMatrix;

1154 bool GWindow::event(QEvent *event)
1155 {
1156  switch (event->type()) {
1157  case QEvent::KeyPress: {
1158  QKeyEvent *key = static_cast<QKeyEvent *>(event);
1159  if (key->key() == Qt::Key_Plus) {
1160  steps = steps + 2;
1161  if (steps < 3) steps = 3;
1162  emit setStatusMessage(QString("Slices: %1").arg(steps));
1163  }
1164  if (key->key() == Qt::Key_Minus) {
1165  steps = steps - 2;
1166  if (steps < 3) steps = 3;
1167  emit setStatusMessage(QString("Slices: %1").arg(steps));
1168  }
1169  if (key->key() == Qt::Key_F) {
1170  if (frame)
1171  frame = false;
1172  else
1173  frame = true;
1174  emit setStatusMessage(QString("Frame: %1").arg(frame));
1175  }
1176  if (key->key() == Qt::Key_T) {
1177  if (trim)
1178  trim = false;
1179  else
1180  trim = true;
1181  emit setStatusMessage(QString("Trim: %1").arg(trim));
1182  }
1183  if (key->key() == Qt::Key_Z) {
1184  rotateXMatrix.setToIdentity();
1185  rotateYMatrix.setToIdentity();
1186  }
1187  if (key->key() == Qt::Key_Y) {
1188  rotateXMatrix.setToIdentity();
1189  rotateXMatrix.rotate(90, 1, 0, 0);
1190  rotateYMatrix.setToIdentity();
1191  }
1192  if (key->key() == Qt::Key_X) {
1193  rotateXMatrix.setToIdentity();
1194  rotateYMatrix.setToIdentity();
1195  rotateYMatrix.rotate(90, 0, -1, 0);
1196  }
1197  if (key->key() == Qt::Key_B) {
1198  rotateXMatrix.setToIdentity();
1199  rotateXMatrix.rotate(180, 0, -1, 0);
1200  rotateYMatrix.setToIdentity();
1201  }
1202  if (key->key() == Qt::Key_G) {
1203  rotateXMatrix.setToIdentity();
1204  rotateXMatrix.rotate(90 + 180, 1, 0, 0);
1205  rotateYMatrix.setToIdentity();
1206  }
1207  if (key->key() == Qt::Key_R) {
1208  rotateXMatrix.setToIdentity();
1209  rotateYMatrix.setToIdentity();
1210  rotateYMatrix.rotate(90 + 180, 0, -1, 0);
1211  }
1212  if (key->key() == Qt::Key_C) {
1213  position.setX(0);
1214  position.setY(0);
1215  }
1216  if (key->key() == Qt::Key_O) {
1217  setOrthogonal(!orthogonal);
1218  }
1219  renderLater();
1220  }
1221  case QEvent::MouseButtonPress:
1222  if (dynamic_cast<QMouseEvent *>(event)->buttons() == Qt::MiddleButton) {
1223  position.setX(0);
1224  position.setY(0);
1225  renderLater();
1226  }
1227  default:
1228  return OpenGLWindow::event(event);
1229  }
