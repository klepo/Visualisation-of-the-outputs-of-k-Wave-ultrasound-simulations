730  }

733  if (this->getRightButtonPressed()) {
734  QPointF lastPosRl = convertPointToOpenGLRelative(this->getLastPositionPressed());
735  QPointF currentPosRl = convertPointToOpenGLRelative(this->getCurrentPositionPressed());

739  }
740 
741  // Zoom
742  if (this->getWheelDelta() > 0)
743  zoom *= 1.0f / 1.1f;
744  else if (this->getWheelDelta() < 0)
745  zoom *= 1.1f;
746 
747  // Projection
748  QMatrix4x4 projectionMatrix;
749  float ratio = float(width()) / float(height());
750 
751  if (orthogonal) {
752  if (zoom > 200.0f) {
753  zoom = 200.0f;
754  }
755  if (zoom < 0.00001f) {
756  zoom = 0.00001f;
757  }
758  projectionMatrix.translate(position.x(), position.y(), 0);
759  projectionMatrix.ortho(-ratio * zoom, ratio * zoom, -zoom, zoom, -1.0f, 1000.0f);
